csharp> string.Format("hex {0}", 0x33)
"hex 51"

csharp> using System.Globalization;

"15x2^2 60"
csharp> string.Format("15x2^-2 {0}", Math.Pow(2, -2) * 15)
"15x2^-2 3.75"
